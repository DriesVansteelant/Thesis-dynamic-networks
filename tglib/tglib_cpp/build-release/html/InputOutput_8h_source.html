103  NodeIdManager nidman;
104  while (getline(fs, line)) {

106  std::vector<Time> l = split_string(line, ' ');
107

109

111  NodeId v = nidman.getNodeId(l[1]);
112

114 
115  Time tt = 1; // todo optional argument
116  if (l.size() >= 4)
117  tt = l[3];
118 
119  edges.push_back({u, v, t, tt});
120  if (!directed) {
121  edges.push_back({v, u, t, tt});
122  }
123 
124  if (t < ti.first) {
125  ti.first = t;
126  }
127 
128  if (t + tt > ti.second) {
129  ti.second = t + tt;
130  }
131  }
132  fs.close();
