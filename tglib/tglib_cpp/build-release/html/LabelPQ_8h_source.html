198  void heapifyUp(size_t p) {
199  size_t child = p;

213  void heapifyDown(size_t p = 0) {
214  size_t length = data.size();
215  while (true) {
216  auto left = getLeftChild(p);
217  auto right = getRightChild(p);
218  auto smallest = p;
219 
220  if (left < length && *(data[left]) < *(data[smallest]))
221  smallest = left;
222 
223  if (right < length && *(data[right]) < *(data[smallest]))
224  smallest = right;
225 
226  if (smallest != p) {
227  swap(smallest, p);
228  p = smallest;
229  } else break;
230  }
