33 inline std::vector<double> temporal_edge_betweenness(tglib::DirectedLineGraph const &dlg) {
34 
35  std::vector<double> betweenness(dlg.num_tg_nodes, 0);
36  std::vector<double> betweenness_dlg(dlg.nodes.size(), 0);
37 
38  // For every node being starting node
39  for (size_t s = 0; s < dlg.nodes.size(); s++) {
40  std::stack<NodeId> S;
41 
42  std::vector<Distance> d(dlg.nodes.size(), -1);
43  d[s] = 0;
44 
45  std::vector<NodeId> prev(dlg.nodes.size(), 0);
46  std::queue<NodeId> Q;
47  Q.push(s);
48 
49  std::vector<double> sigma(dlg.nodes.size(), 0);
50  sigma[s] = 1;
51 
52  std::map<NodeId, std::vector<NodeId>> P; // neighbours visited
53 
54  // Brandes' alg.
55  while(!Q.empty()) {
56  int u = Q.front();
57  Q.pop();
58  S.push(u);
59 
60  for (auto &e : dlg.nodes[u].outedges) {
61  NodeId v = e.v;
62  if(d[v] < 0) {
63  Q.push(v);
64  d[v] = d[u] + 1;
65  }
66 
67  // Update sigma if shortest path
68  if(d[v] == d[u] + 1) {
69  sigma[v] = sigma[v] + sigma[u];
70  P[v].push_back(u);
71  }
72  }
73  }
74  std::vector<double> delta(dlg.nodes.size(), 0);
75 
76  while (!S.empty()) {
77  int w = S.top();
78  S.pop();
79  for(auto it = P[w].begin(); it != P[w].end(); ++it) {
80  int v = *it;
81  delta[v] += ((sigma[v]/sigma[w])*(1+ delta[w]));
82  }
83  if ((size_t)w != s) {
84  betweenness_dlg[w] += delta[w];
85  }
86  }
87  }
88 
89  return betweenness_dlg;
90 }
