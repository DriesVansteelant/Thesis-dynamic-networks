40 double temporal_clustering_coefficient(IncidentLists<N, E> const& tg, NodeId nid, TimeInterval ti){
41 
42  std::set<NodeId> neighbors;
43  for (auto &e : tg.getNode(nid).outEdges) {
44  if (e.t < ti.first || e.t > ti.second) continue;
45  neighbors.insert(e.v);
46  }
47 
48  double count = 0;
49  for (auto &v : neighbors) {
50  for (auto &e : tg.getNode(v).outEdges) {
51  if (neighbors.find(e.v) != neighbors.end()) {
52  count += 1;
53  }
54  }
55  }
56 
57  if (count == 0) return 0;
58 
59  auto timesteps = ti.second - ti.first;
60  auto m = (double)(neighbors.size() * (neighbors.size() - 1));
61  double result = (1.0 / (double)timesteps) * (count / m);
62 
63  return result;
64 }
