36 std::vector<double> temporal_pagerank(OrderedEdgeList<E> const &tgs, double alpha, double beta, double gamma) {
37 
38  std::vector<double> degrees(tgs.getNumberOfNodes(), 0.0);
39  for (auto &e : tgs.getEdges()) {
40  degrees.at(e.u)++;
41  }
42  for (auto &d : degrees) d /= tgs.getEdges().size();
43 
44  double normalization = 1.0;
45  if (beta == 1.0) beta = 0.0;
46 
47  std::vector<double> r(tgs.getNumberOfNodes(), 0.0);
48  std::vector<double> s(tgs.getNumberOfNodes(), 0.0);
49 
50  for (auto &e : tgs.getEdges()) {
51  r[e.u] = r[e.u] * gamma + 1.0 * (1.0 - alpha) * degrees[e.u] * alpha * normalization;
52  r[e.v] = r[e.v] * gamma + s[e.u] + 1.0 * (1.0 - alpha) * degrees[e.u] * alpha * normalization;
53  s[e.v] = s[e.v] + (s[e.u] + 1.0 * (1.0 - alpha)* s[e.u]) * alpha * (1.0 - beta);
54  s[e.u] = s[e.u] * beta;
55  }
56 
57  return r;
58 }
