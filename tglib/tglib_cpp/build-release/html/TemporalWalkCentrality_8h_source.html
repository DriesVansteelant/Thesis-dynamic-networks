29 std::vector<std::map<Time, double>> getIncomingWeightedWalks(OrderedEdgeList<E> const &tgs, double alpha) {
30  std::vector<std::map<Time, double>> walks(tgs.getNumberOfNodes(), std::map<Time, double>());
31  for (auto &e: tgs.getEdges()) {
32  walks.at(e.v)[e.t + 1] += 1; //alpha;
33  for (auto &x: walks.at(e.u)) {
34  if (x.first > e.t) {
35  continue;
36  }
37  walks.at(e.v)[e.t + 1] += (alpha * x.second);
38  }
39  }
40  return walks;
41 }

50 template<typename E>
51 std::vector<std::map<Time, double>> getOutgoingWeightedWalks(OrderedEdgeList<E> const &tgs, double beta) {
52  std::vector<std::map<Time, double>> walks(tgs.getNumberOfNodes(), std::map<Time, double>());
53  for (unsigned long ep = tgs.getEdges().size() - 1;; --ep) {
54  auto &e = tgs.getEdges()[ep];
55  walks.at(e.u)[e.t] += 1; //beta;
56  for (auto &x: walks.at(e.v)) {
57  if (x.first < e.t + 1) {
58  continue;
59  }
60  walks.at(e.u)[e.t] += (beta * x.second);
61  }
62  if (ep == 0) break;
63  }
64  return walks;
65 }

77 std::vector<double>
78 computeCentrality(OrderedEdgeList<E> const &tgs, std::vector<std::map<Time, double>> &ifct, std::vector<std::map<Time, double>> &ofct) {
79  std::vector<double> centrality(tgs.getNumberOfNodes(), 0);
80 
81 #pragma omp parallel for default(none) shared(centrality, tgs, ifct, ofct)
82  for (size_t nid = 0; nid < tgs.getNumberOfNodes(); ++nid) {
83 
84  auto init = ifct[nid].begin();
85  auto outit = ofct[nid].begin();
86 
87  double insum = 0;
88 
89  while (outit != ofct[nid].end()) {
90  if (init != ifct[nid].end() && outit != ofct[nid].end() && init->first < outit->first) {
91  insum += init->second;
92  ++init;
93  }
94 
95  if (init != ifct[nid].end() && outit != ofct[nid].end() && outit->first < init->first) {
96  if (init->first <= outit->first)
97  insum += init->second;
98  centrality[nid] += insum * outit->second;
99  ++outit;
100  }
101  if (init != ifct[nid].end() && outit != ofct[nid].end() && init->first == outit->first) {
102  if (init->first <= outit->first)
103  insum += init->second;
104  centrality[nid] += insum * outit->second;
105  ++init;
106  ++outit;
107  }
108 
109  if (init == ifct[nid].end() && outit != ofct[nid].end()) {
110  while (outit != ofct[nid].end()) {
111  centrality[nid] += insum * outit->second;
112  ++outit;
113  }
114  break;
115  }
116  }
117  }
118  return centrality;
119 }
