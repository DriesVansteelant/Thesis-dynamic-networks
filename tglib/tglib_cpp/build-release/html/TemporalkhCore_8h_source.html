29 std::vector<uint> compute_kcores(std::vector<StaticWeightedEdge> const &edges) {
30  std::map<uint, std::vector<StaticWeightedEdge>> g;
31 
32  NodeId mxnid = 0;
33  for (auto &e : edges) {
34  g[e.u].push_back(e); // todo assumption graph is undirected
35  if (e.u > mxnid) mxnid = e.u;
36  if (e.v > mxnid) mxnid = e.v;
37  }
38 
39  std::set<std::pair<uint, NodeId>, std::less<>> degrees;
40  std::vector<uint> c(mxnid+1, 0);
41 // std::vector<uint> c(g.size(), 0);
42  for (auto &p : g) {
43  degrees.insert({p.second.size(), p.first});
44  c[p.first] = p.second.size();
45  }
46 
47  std::vector<bool> removed(mxnid+1, false);
48 
49  while (!degrees.empty() ) {
50 
51  // get next node
52  auto next = degrees.begin();
53  auto u = next->second;
54  degrees.erase(next);
55 
56  if (removed[u]) continue;
57  removed[u] = true;
58 
59  for (auto &e : g[u]) {
60  if (removed[e.v]) continue;
61  if (c[e.v] > c[e.u]) {
62  --c[e.v];
63  degrees.insert({c[e.v], e.v});
64  }
65  }
66  }
67 
68  return c;
69 }
