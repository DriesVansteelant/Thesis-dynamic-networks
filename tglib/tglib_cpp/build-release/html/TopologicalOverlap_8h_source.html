47 double topological_overlap(IncidentLists<N, E> const& tg, NodeId nid, TimeInterval ti) {
48  std::vector<TemporalEdge> edges = tg.getNode(nid).outEdges;
49  if (edges.empty()) return 0;
50 
51  std::sort(edges.begin(), edges.end());
52 
53  std::vector<std::set<NodeId>> counts;
54  Time curTime = edges[0].t;
55  counts.emplace_back();
56  for (auto &e : edges) {
57  if (e.t < ti.first) continue;
58  if (e.t > ti.second) break;
59  if (e.t != curTime) {
60  curTime = e.t;
61  counts.emplace_back();
62  }
63  counts.back().insert(e.v);
64  }
65 
66  double to = 0;
67  for (size_t i = 0; i < counts.size()-1; ++i) {
68  std::vector<int> v(std::min(counts[i].size(), counts[i+1].size()));
69  auto it = std::set_intersection(counts[i].begin(), counts[i].end(),
70  counts[i+1].begin(), counts[i+1].end(), v.begin());
71  v.resize(it-v.begin());
72  auto d = (double) v.size();
73  auto n = (double) std::sqrt(counts[i].size() * counts[i+1].size());
74 
75  to += (d/n);
76  }
77  to /= ((double)counts.size() - 1.0);
78  return to;
79 }
