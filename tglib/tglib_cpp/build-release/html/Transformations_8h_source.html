72 template<typename E>
73 tglib::TRSGraph to_trs_graph(tglib::OrderedEdgeList<E> const &tgs) {
74  TRSGraph trs;
75  trs.positionOfEarliestTimeNode.resize(tgs.getNumberOfNodes(), inf);
76  trs.positionOfLatestTimeNode.resize(tgs.getNumberOfNodes(), inf);
77  trs.numTGNodes = tgs.getNumberOfNodes();
78  trs.numEdges = 0;
79 
80  std::set<TimeNode> timeNodes;
81  std::vector<Time> maxArrivalTimes(tgs.getNumberOfNodes(), -1);
82  std::vector<std::set<Time>> startingTimes(tgs.getNumberOfNodes(), std::set<Time>());
83 
84  std::map<TimeNode, NodeId> nodeMap;
85 
86  NodeId nid = 0;
87 
88  for (auto &e: tgs.getEdges()) {
89  startingTimes[e.u].insert(e.t);
90  if (maxArrivalTimes[e.v] < e.t + e.tt) {
91  maxArrivalTimes[e.v] = e.t + e.tt;
92  }
93  }
94  for (size_t i = 0; i < tgs.getNumberOfNodes(); ++i) {
95  if (maxArrivalTimes[i] > -1) {
96  timeNodes.insert({i, maxArrivalTimes[i]});
97  startingTimes[i].insert(maxArrivalTimes[i]);
98  }
99 
100  auto numberOfStartingTimes = startingTimes[i].size();
101  for (Time startingTime: startingTimes[i]) {
102  TRSNode trsNode;
103  trsNode.tn = {i, startingTime};
104  trsNode.nid = nid++;
105 
106  timeNodes.insert(trsNode.tn);
107  nodeMap[trsNode.tn] = trsNode.nid;
108 
109  if (numberOfStartingTimes > 1) {
110  TRSEdge trsEdge{};
111  trsEdge.u = trsNode.nid;
112  trsEdge.v = nid;
113  trsEdge.tt = 0;
114  trsNode.adjList.push_back(trsEdge);
115  trs.numEdges++;
116  }
117  trs.nodes.push_back(trsNode);
118 
119  if (numberOfStartingTimes == startingTimes[i].size()) {
120  trs.positionOfEarliestTimeNode[i] = trs.nodes.size() - 1;
121  }
122 
123  numberOfStartingTimes--;
124 
125  if (numberOfStartingTimes == 0) {
126  trs.positionOfLatestTimeNode[i] = trs.nodes.size() - 1;
127  }
128 
129  }
130  }
131  for (auto &e: tgs.getEdges()) {
132  NodeId u = nodeMap[{e.u, e.t}];
133  auto it = lower_bound(startingTimes[e.v].begin(), startingTimes[e.v].end(), e.t + e.tt);
134  assert(it != startingTimes[e.v].end());
135  Time arrTime = *(it);
136  NodeId v = nodeMap[{e.v, arrTime}];
137  TRSEdge trsEdge{u, v, e.tt};
138  trs.nodes[u].adjList.push_back(trsEdge);
139  trs.numEdges++;
140  }
141 
142  trs.ti = tgs.getTimeInterval();
143 
144  return trs;
145 }

153 template<typename N, typename E>
154 tglib::DirectedLineGraph to_directed_line_graph(tglib::IncidentLists<N, E> const &tg) {
155  DirectedLineGraph dlg;
156 
157  dlg.nodes.clear();
158  dlg.nodes.resize(tg.getNumberOfEdges());
159  std::vector<bool> nodeset(tg.getNumberOfEdges(), false);
160 
161  std::map<TemporalEdge, EdgeId> eids;
162  EdgeId eid = 0;
163  for (auto &tgn: tg.getNodes()) {
164  for (auto &e: tgn.outEdges) {
165  eids[e] = eid++;
166  }
167  }
168 
169  for (auto &tgn: tg.getNodes()) {
170  for (auto &e: tgn.outEdges) {
171  if (!nodeset.at(eids[e])) {
172  DLNode node;
173  node.id = eids[e];
174  node.e = e;
175  dlg.nodes.at(node.id) = node;
176  nodeset.at(eids[e]) = true;
177  }
178  for (auto &f: tg.getNode(e.v).outEdges) {
179  if (e.t + e.tt > f.t) continue;
180  if (!nodeset.at(eids[f])) {
181  DLNode node;
182  node.id = eids[f];
183  node.e = f;
184  dlg.nodes.at(node.id) = node;
185  nodeset.at(eids[f]) = true;
186  }
187  auto &m = dlg.nodes.at(eids[f]);
188  auto &n = dlg.nodes.at(eids[e]);
189  auto de = DLEdge();
190  de.u = n.id;
191  de.v = m.id;
192  de.u_t = n.e.t;
193  de.v_t = m.e.t;
194  de.tg_u = n.e.u;
195  de.tg_w = m.e.v;
196  de.tg_v = n.e.v;
197  n.outedges.push_back(de);
198 // m.inedges.push_back(de);
199  dlg.num_edges++;
200  }
201  }
202  }
203  dlg.num_tg_nodes = tg.getNumberOfNodes();
204 
205  return dlg;
206 }

255 template<typename E>
256 OrderedEdgeList<E> normalize(tglib::OrderedEdgeList<E> const &tgs, bool removeLoops) {
257  std::set<E> edges;
258  for (auto &e: tgs.getEdges()) {
259  if (removeLoops && e.u == e.v) continue;
260  edges.insert(e);
261  }
262  std::vector<E> filtered_edges;
263  TimeInterval ti = {inf, 0};
264  for (auto &e: edges) {
265  filtered_edges.push_back(e);
266  if (e.t < ti.first) ti.first = e.t;
267  if (e.t + e.tt > ti.second) ti.second = e.t + e.tt;
268  }
269  OrderedEdgeList<E> normalized_graph(tgs.getNumberOfNodes(), filtered_edges, ti, tgs.getNodeMap());
270  return normalized_graph;
271 }

332 template<typename N, typename E>
333 tglib::IncidentLists<N, E> reverse(tglib::IncidentLists<N, E> &tg, tglib::TimeInterval ti) {
334  std::vector<E> tes;
335  for (TGNode const &n: tg.getNodes()) {
336  for (TemporalEdge const &e: n.outEdges) {
337  if (e.t < ti.first || e.t + e.tt > ti.second) continue;
338  TemporalEdge x{};
339  x.v = e.u;
340  x.u = e.v;
341  x.t = 1 + ti.second - e.t;
342  x.tt = 1;
343  tes.push_back(x);
344  }
345  }
346  std::sort(tes.begin(), tes.end());
347  tglib::IncidentLists<N, E> rtg(tg.getNumberOfNodes(), tes);
348  return rtg;
349 }
